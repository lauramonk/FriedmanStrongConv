\begin{definition}
  \label{def:non_comm_poly}
  \uses{def:word}
  Let $D \geq 1$ and $r \geq 1$.
  A \emph{non-commutative polynomial with matrix coefficients} $P \in \nonco$
  is an expression of the form $\sum_{w \in \word} A_w \otimes w(x_1, \ldots, x_{r})$.
\end{definition}

  \label{def:non_co_map}
  \uses{def:non_comm_poly,def:word_map}
  Let $P = \sum_{w \in \word} A_w \otimes w(x_1, \ldots, x_{r})\in \nonco$.
  For $(U_1, \ldots, U_r)$ be a $r$-tuple of unitary matrices of shared dimension,
  we define
  $P(U_1, \ldots, U_r) = \sum_{w \in \word} A_w \otimes w(U_1, \ldots, U_r)$
  using the word map defined in Definition \ref{def:word_map}.
\end{definition}

\section{The Master inequality for polynomials}

We are now ready to prove the following statement.

\begin{theorem}
  \label{thm:masterI}
  \uses{def:self_adjoint_P,def:probability_measure}
  \notready
  Let $r \geq 2$ and $D \geq 1$.
  Fix a self-adjoint non-commutative polynomial $P \in \nonco$
  of degree $q_0$, and let $K = \|P\|_{D,r}$.
  There exists a family of linear functionals $\nu_i : \mathcal{P} \rightarrow \C$
  for $i \geq 0$ such that for every $N,m,q \geq 1$, and any $h \in \mathcal{P}_q$,
  \begin{equation*}
    \left|\expec\left[\trN[ND] h(P(S_1, \ldots, S_r))\right]
     - \sum_{i=0}^{m-1} \frac{\nu_i(h)}{N^i}\right|
     \leq \frac{(4 q q_0(1+\log r))^{4m}}{N^m} \|h\|_{C^0([-K,K])}.
  \end{equation*}
\end{theorem}

\subsection{Bound on the coefficients}

We now prove a bound on the coefficients of the expansion.

\begin{lemma}
  \label{coro:masterI}
  \uses{lem:masterI_largeN,lem:expr_nu_0,lem:bound_trace,lem:expansion_bound}
  With the notations of Theorem \ref{thm:masterI},
  for every $m, q \geq 1$ and $h \in \mathcal{P}_q$,
  \begin{equation*}
    |\nu_m(h)|
    \leq (4 q q_0(1+\log r))^{4m} \|h\|_{C^0([-K,K])}.
  \end{equation*}
\end{lemma}

This relies on the following simple lemma on asymptotic expansions.

\begin{lemma}
  \label{lem:expansion_bound}
  Let $(F_N)_{N \geq 0}$ be a sequence.
  We assume there exists coefficients $(\nu_i)_{i \geq 0}$, constants $(C_m)_{m \geq 1}$ and
  $(M_m)_{m \geq 1}$ such that, for all $m \geq 1$, all $N \geq M_m$,
  \begin{equation*}
    \left| F_N - \sum_{i=0}^{m-1} \frac{\nu_i}{N^i} \right| \leq \frac{C_m}{N^m}.
  \end{equation*}
  Then, for all $m \geq 1$, $|\nu_m| \leq C_m$.
\end{lemma}

\begin{proof}
  Let $m \geq 1$ and $N \geq \max(M_m, M_{m+1})$.
  We have
  \begin{equation*}
    \nu_m = N^m \left(F_N - \sum_{i=0}^{m-1} \frac{\nu_i}{N^i} \right)
    - N^m \left(F_N - \sum_{i=0}^{m} \frac{\nu_i}{N^i}\right)
  \end{equation*}
  so by the triangle inequality,
    \begin{equation*}
    |\nu_m| \leq N^m \left|F_N - \sum_{i=0}^{m-1} \frac{\nu_i}{N^i} \right|
    + N^m \left|F_N - \sum_{i=0}^{m} \frac{\nu_i}{N^i}\right|.
  \end{equation*}
  We apply the hypothesis to $m$ and $m+1$ (which we can as $N \geq M_m$ and $N \geq M_{m+1}$),
  which yields
  \begin{equation*}
    |\nu_m|
    \leq N^m \frac{C_m}{N^m} + N^m \frac{C_{m+1}}{N^{m+1}}
    = C_{m} + \frac{C_{m+1}}{N}.
  \end{equation*}
  This is true for all $N \geq \max(M_m,M_{m+1})$, and hence we can take the limit $N \rightarrow + \infty$,
  and obtain $|\nu_m| \leq C_m$.
\end{proof}

We will also need to bound the $0$-th coefficient.

\begin{lemma}
  \label{lem:expr_nu_0}
  \uses{lem:masterI_largeN}
  For any $h \in \mathcal{P}$,
  \begin{equation*}
    \nu_0(h) = \lim_{N \rightarrow +\infty} \expec [\trN[ND] h(P(S_1, \ldots, S_r))].
  \end{equation*}
\end{lemma}

\begin{proof}
  Let $h \in \mathcal{P}$ and take $q \geq 1$ such that $h \in \mathcal{P}_q$.
  We apply Lemma \ref{lem:masterI_largeN} for $m=1$.
  Hence, for any $N \geq M$,
  \begin{equation*}
    \left|
      \expec [\trN[ND] h(P(S_1, \ldots, S_r))]
      - \nu_0(h)
    \right| \leq \frac{(4 q q_0(1+\log r))^4}{N}
  \end{equation*}
  The right-hand side goes to $0$ as $N \rightarrow \infty$, which leads to the claim.
\end{proof}

We can finally prove Lemma \ref{coro:masterI}.

\begin{proof}
  \proves{coro:masterI}
  Let $m \geq 0$, $q \geq 1$ and $h \in \mathcal{P}_q$.

  If $m=0$, by Lemma \ref{lem:expr_nu_0},
  \begin{equation*}
        \nu_0(h) = \lim_{N \rightarrow +\infty} \expec [\trN[ND] h(P(S_1, \ldots, S_r))].
  \end{equation*}
  By Lemma \ref{lem:bound_trace}, this sequence is bounded by $\|h\|_{C^0([-K,K])}$,
  and in particular we get that at the limit
  $|\nu_0(h)| \leq \|h\|_{C^0([-K,K])}$. This is the claim for $m=0$.

  The result for $m \geq 1$ is a direct combination of Lemma \ref{lem:masterI_largeN} and
  Lemma \ref{lem:expansion_bound}.
\end{proof}

\section{Special cases}

\begin{lemma}
  \label{lem:value_nu_at_one}
  \uses{thm:masterI}
  With the notations of Theorem \ref{thm:masterI},
  if $h$ is a constant polynomial, then
  \begin{itemize}
    \item $\nu_0(h)=h(0)$;
    \item for $i >0$, $\nu_i(h)=0$.
  \end{itemize}
\end{lemma}

\begin{proof}
  If $h$ is constant, then for any $N \geq 1$,
  $\expec\left[\trN[ND] h(P(S_1, \ldots, S_r))\right] = h(0)$
  and the sequence from the claim satisfies the inequality from Theorem \ref{thm:masterI} and
  is hence the value of $(\nu_i(h))_{i \geq 0}$ by uniqueness [todo].
\end{proof}
